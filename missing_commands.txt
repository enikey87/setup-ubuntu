# Generate an SSH key 
ssh-keygen -t ed25519 -C "enikey87@gmail.com" -f ~/.ssh/id_ed25519 -N ""
# Add your SSH key to the ssh-agent
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_ed25519
cat ~/.ssh/id_ed25519.pub
